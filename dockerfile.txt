# Build stage
FROM golang:1.21-alpine AS builder

# Install build dependencies
RUN apk add --no-cache git make

# Set working directory
WORKDIR /build

# Copy go mod files
COPY go.mod go.sum ./
RUN go mod download

# Copy source code
COPY . .

# Build the binary
RUN make build-bridge

# Runtime stage
FROM alpine:3.19

# Install runtime dependencies
RUN apk add --no-cache \
    ca-certificates \
    docker-cli \
    git \
    curl \
    bash

# Install act
RUN curl -sL https://github.com/nektos/act/releases/download/v0.2.65/act_Linux_x86_64.tar.gz | \
    tar xz -C /usr/local/bin act && \
    chmod +x /usr/local/bin/act

# Create non-root user
RUN addgroup -g 1000 -S actionsbridge && \
    adduser -u 1000 -S actionsbridge -G actionsbridge

# Create directories
RUN mkdir -p /var/lib/actions-bridge /etc/actions-bridge && \
    chown -R actionsbridge:actionsbridge /var/lib/actions-bridge /etc/actions-bridge

# Copy binary from builder
COPY --from=builder /build/bin/actions-bridge /usr/local/bin/actions-bridge

# Copy default configuration
COPY --chown=actionsbridge:actionsbridge docker/config.yaml /etc/actions-bridge/config.yaml

# Switch to non-root user
USER actionsbridge

# Set environment variables
ENV ACTIONS_BRIDGE_BASE_DIR=/var/lib/actions-bridge \
    ACT_DEFAULT_IMAGE=catthehacker/ubuntu:act-latest

# Expose health check port
EXPOSE 8080

# Health check
HEALTHCHECK --interval=30s --timeout=10s --start-period=10s --retries=3 \
    CMD curl -f http://localhost:8080/health || exit 1

# Entry point
ENTRYPOINT ["/usr/local/bin/actions-bridge"]
